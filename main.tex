% 2. Modificamos los comandos para que el margen izquierdo se aplique AUTOMÁTICAMENTE después del título
\let\oldsection\section
\let\oldsubsection\subsection
\let\oldsubsubsection\subsubsection

\DeclareDocumentCommand{\section}{ s o m }{%
  \IfBooleanTF{#1}{\oldsection*{#3}}{\IfValueTF{#2}{\oldsection[#2]{#3}}{\oldsection{#3}}}%
  \par\setlength{\leftskip}{0pt}%
}

\DeclareDocumentCommand{\subsection}{ s o m }{%
  \IfBooleanTF{#1}{\oldsubsection*{#3}}{\IfValueTF{#2}{\oldsubsection[#2]{#3}}{\oldsubsection{#3}}}%
  \par\setlength{\leftskip}{1.27cm}%
}

\DeclareDocumentCommand{\subsubsection}{ s o m }{%
  \IfBooleanTF{#1}{\oldsubsubsection*{#3}}{\IfValueTF{#2}{\oldsubsubsection[#2]{#3}}{\oldsubsubsection{#3}}}%
  \par\setlength{\leftskip}{2.54cm}%
}
